    private fun formatDisplayTempFromF(tempF: Float?): String {
        if (tempF == null) return "--"
        return if (useFahrenheit) {
            "${tempF.toInt()}°F"
        } else {
            "${((tempF - 32f) * 5f / 9f).toInt()}°C"
        }
    }
